### Array.prototype.includes()

> check whether array includes that spcify value, return true if included else return false

```js
arr.includes(x);
arr.indexOf(x) >= 0;
```

> before es7

```js
let arr = ["a", "b", "c"];

if (arr.indexOf("a") !== -1) {
  console.log("a exists");
}
```

> with es7

```js
let arr = ["a", "b", "c"];

if (arr.includes("a")) {
  console.log("a exists");
}
```
